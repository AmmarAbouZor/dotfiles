# Config to add to ~/.gitconfig file

# fast forward on pull only
[pull]
	ff = only

[diff]
	tool = nvim

[difftool "nvim"]
   cmd = nvim -d "$LOCAL" "$REMOTE" 

# Nvim as merge tool `git mergetool`
[merge]
	tool = nvim

[mergetool "nvim"]
   cmd = nvim -d "$LOCAL" "$MERGED" "$REMOTE"
